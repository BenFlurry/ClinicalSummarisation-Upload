<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title>PDF Guidance Viewer</title>
    <script src="pdf.js"></script> 
    <style>
        body {
            margin: 0;
            background-color: #f1f5f9;
            display: flex;
            justify-content: center;
            padding: 20px;
        }
        #canvas-container {
            position: relative;
            box-shadow: 0 4px 6px rgba(0, 0, 0, 0.1);
        }
        /* The yellow highlight box overlay */
        .highlight {
            position: absolute;
            background-color: rgba(250, 204, 21, 0.4); /* Yellow with opacity */
            border: 2px solid #eab308;
            border-radius: 4px;
            pointer-events: none; /* Let user click through it */
            display: none;
        }
    </style>
</head>
<body>

    <div id="canvas-container">
        <canvas id="pdf-canvas"></canvas>
        <div id="highlight-box" class="highlight"></div>
    </div>

    <script>
        // 1. Setup PDF.js worker
        pdfjsLib.GlobalWorkerOptions.workerSrc = 'pdf.worker.js';

        // 2. Parse parameters from the WebView2 navigation URI
        const urlParams = new URLSearchParams(window.location.search);
        const fileUrl = urlParams.get('file');
        const pageNum = parseInt(urlParams.get('page')) || 1;
        
        const x0 = parseFloat(urlParams.get('x0'));
        const y0 = parseFloat(urlParams.get('y0'));
        const x1 = parseFloat(urlParams.get('x1'));
        const y1 = parseFloat(urlParams.get('y1'));

        if (fileUrl) {
            pdfjsLib.getDocument(fileUrl).promise.then(function(pdf) {
                return pdf.getPage(pageNum);
            }).then(function(page) {
                // Set scale for high DPI / readability
                const scale = 1.5; 
                const viewport = page.getViewport({scale: scale});

                const canvas = document.getElementById('pdf-canvas');
                const context = canvas.getContext('2d');
                canvas.height = viewport.height;
                canvas.width = viewport.width;

                const renderContext = {
                    canvasContext: context,
                    viewport: viewport
                };

                // 3. Render the PDF onto the canvas
                page.render(renderContext).promise.then(function() {
                    // 4. Draw the highlight if coordinates exist
                    if (!isNaN(x0) && !isNaN(y0) && !isNaN(x1) && !isNaN(y1)) {
                        const highlight = document.getElementById('highlight-box');
                        
                        // Calculate width and height
                        let width = x1 - x0;
                        let height = y1 - y0;

                        // Scale coordinates to match the canvas zoom level
                        let scaledX = x0 * scale;
                        let scaledY = y0 * scale;
                        let scaledWidth = width * scale;
                        let scaledHeight = height * scale;

                        // NOTE: If your RAG extractor uses standard PDF origin (bottom-left = 0,0), 
                        // uncomment the next line to invert the Y-axis for HTML (top-left = 0,0):
                        // scaledY = viewport.height - (y1 * scale); 

                        // Apply to the overlay div
                        highlight.style.left = scaledX + 'px';
                        highlight.style.top = scaledY + 'px';
                        highlight.style.width = scaledWidth + 'px';
                        highlight.style.height = scaledHeight + 'px';
                        highlight.style.display = 'block';

                        // Scroll into view
                        highlight.scrollIntoView({ behavior: 'smooth', block: 'center' });
                    }
                });
            }).catch(function(error) {
                console.error('Error rendering PDF:', error);
            });
        }
    </script>
</body>
</html>
